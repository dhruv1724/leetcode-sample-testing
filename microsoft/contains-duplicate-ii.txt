class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int,int>foundAt;
        int diff=0;
        for(int i=0;i<nums.size();i++){
            if(foundAt.find(nums[i])!=foundAt.end()){
                diff=abs(foundAt[nums[i]]-i);
                if(diff<=k) return true;
                foundAt[nums[i]]=i;
            }else{
                foundAt[nums[i]]=i;
            }
        }
        return false;
    }
};
